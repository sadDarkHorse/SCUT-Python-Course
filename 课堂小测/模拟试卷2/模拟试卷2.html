(本题分数:2)
　　★【附件下载】★

【参考答案】
f=open("2015年5月高三模拟考成绩.csv","r",encoding='utf8')
score={}
for line in f:
    a=line.split(",")
    if a[0][0]=='3':
        b=score.get(a[1],[0,0])  #把总分，人数作为列表，保存到字典
        b[0]=b[0]+eval(a[4])
        b[1]=b[1]+1
        score[a[1]]=b
f.close()

#按平均分排序
ls=sorted(score.items(),key=lambda x:-x[1][0]/x[1][1])

print("名次\t班别\t数学平均分")

for i in range(len(ls)):
    print("{:3}\t{}\t{:.2f}".format(i+1,ls[i][0],ls[i][1][0]/ls[i][1][1]))
